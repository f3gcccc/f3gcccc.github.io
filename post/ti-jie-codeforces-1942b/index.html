题解：CodeForces 1942B

2024-09-26

可以从第一个数开始看。

mex1=0or1mex_{1} = 0 or 1mex1​=0or1，接着往后推就行了。

若ai>0,则有mexi−pi>0若 a_{i}>0, 则有 mex_{i}-p_{i}>0若ai​>0,则有mexi​−pi​>0 说明 mexmexmex 变大了，那唯一的情况就是 pip_{i}pi​ 把原来的 mexmexmex 覆盖掉了

若ai<0,有mexi<pi,也就是mex没变,可根据式子mex−pi=ai移个项就变成了pi=mexi−ai若 a_{i}<0, 有 mex_{i}<p_{i}, 也就是 mex 没变, 可根据式子 mex-p_{i}=a_{i} 移个项就变成了 p_{i}=mex_{i}-a_{i}若ai​<0,有mexi​<pi​,也就是mex没变,可根据式子mex−pi​=ai​移个项就变成了pi​=mexi​−ai​

代码中的 b[] 即为 ppp

#include <bits/stdc++.h>
#define ll long long
using namespace std;
int a[200005],b[200005];
bool vis[200005];
void solve()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++) cin>>a[i];
	if(a[1]==1) b[1]=0;
	else b[1]=(-a[1]);
	int mex;
	if(b[1]!=0) mex=0;
	else mex=1;
	memset(vis,0,sizeof(vis));
	vis[b[1]]=1;
	for(int i=2;i<=n;i++)
	{
		if(a[i]>0) b[i]=mex;
		else b[i]=mex-a[i];
		vis[b[i]]=1;
		while(vis[mex]) mex++;
	}
	for(int i=1;i<=n;i++) cout<<b[i]<<' ';
	cout<<'\n';
}
int main()
{
	int t;
	cin>>t;
	while(t--) solve();
	return 0;
}
